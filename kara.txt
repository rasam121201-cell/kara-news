  <script>
    // small JS helpers: year, simple form validation + demo "send"
    document.getElementById('year').textContent = new Date().getFullYear();

    function sleep(ms){return new Promise(res=>setTimeout(res,ms))}

    async function handleSubmit(e){
      e.preventDefault();
      const name = document.getElementById('name').value.trim();
      const email = document.getElementById('email').value.trim();
      const message = document.getElementById('message').value.trim();
      const status = document.getElementById('formStatus');

      if(!name || !email || !message){
        status.textContent = 'Please fill required fields.';
        return false;
      }

      // fake send demo
      status.textContent = 'Sending...';
      await sleep(800);
      status.textContent = 'Message sent — thanks! (demo)';

      // reset form after a moment
      await sleep(800);
      e.target.reset();
      setTimeout(()=>status.textContent = "We won't spam you — promise.", 800);
      return false;
    }
  </script>
